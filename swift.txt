
extension String {
  var isNumeric: Bool {
    return Double(self) != nil
  }

  func withPrefix(_ prefix: String) -> String {
    return prefix + self
  }

  var lines: [String] {
    return self.components(separatedBy: "\n")
  }
}

print("4i00".isNumeric)

print("pet".withPrefix("car"))


print("this\nis\na\ntest".lines)
